67class GreenBreath : public RgbLedPattern {
68 public:
69 GreenBreath() = default;
70
71 Rgb Update(TimePoint now) override {
72 auto elapsed = ElapsedMs(start_time_, now);
73 // 呼吸周期：1700ms
74 constexpr u32 kBreathPeriod = 1700;
75 u32 phase = elapsed % kBreathPeriod;
76
77 f32 brightness;
78 if (phase < kBreathPeriod / 2) {
79 // 前半周期：从0到255
80 brightness = (phase * 255.0f) / (kBreathPeriod / 2);
81 } else {
82 // 后半周期：从255到0
83 brightness = 255.0f - ((phase - kBreathPeriod / 2) * 255.0f) / (kBreathPeriod / 2);
84 }
85
86 return Rgb(0, static_cast<u8>(brightness), 0);
87 }
88
89 void Reset(TimePoint now) override { start_time_ = now; }
90
91 private:
92 TimePoint start_time_;
93};

99class RedFlash : public RgbLedPattern {
100 public:
101 RedFlash() = default;
102
103 Rgb Update(TimePoint now) override {
104 auto elapsed = ElapsedMs(start_time_, now);
105 // 闪烁周期：1000ms
106 u32 phase = elapsed % 1000;
107 if (phase < 500) {
108 return Rgb(255, 0, 0);
109 }
110 return Rgb(0, 0, 0);
111 }
112
113 void Reset(TimePoint now) override { start_time_ = now; }
114
115 private:
116 TimePoint start_time_;

123class RgbFlow : public RgbLedPattern {
124 public:
125 RgbFlow() = default;
126
127 Rgb Update(TimePoint now) override {
128 auto elapsed = ElapsedMs(start_time_, now);
129 // 完整颜色循环周期：1536ms
130 u32 phase = elapsed % 1536;
131
132 u8 r, g, b;
133 if (phase < 512) {
134 r = 255 - (phase * 255) / 512;
135 g = (phase * 255) / 512;
136 b = 0;
137 } else if (phase < 1024) {
138 r = 0;
139 g = 255 - ((phase - 512) * 255) / 512;
140 b = ((phase - 512) * 255) / 512;
141 } else {
142 r = ((phase - 1024) * 255) / 512;
143 g = 0;
144 b = 255 - ((phase - 1024) * 255) / 512;
145 }
146 return Rgb(r, g, b);
147 }
148
149 void Reset(TimePoint now) override { start_time_ = now; }
150
151 private:
152 TimePoint start_time_;
153};
