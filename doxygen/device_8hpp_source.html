97template <size_t kMaxDevices, bool kUseStdFunctionCallback = true>
98class DeviceManager {
99 using CallbackType = //
100 std::conditional_t<kUseStdFunctionCallback, //
101 std::function<void(Device *)>, //
102 void (*)(Device *)>;
103
104 public:
105 DeviceManager() = default;
106
107 DeviceManager(std::initializer_list<Device *> devices) {
108 if (devices.size() <= kMaxDevices) {
109 devices_ = devices;
110 } else {
111 Throw(std::runtime_error("DeviceManager: Too many devices!"));
112 }
113 }
114
115 DeviceManager &operator<<(Device *device) {
116 if (devices_.size() < kMaxDevices) {
117 devices_.push_back(device);
118 } else {
119 Throw(std::runtime_error("DeviceManager: Too many devices!"));
120 }
121 return *this;
122 }
123
128 bool Update() {
129 all_device_ok_ = true;
130 for (auto device : devices_) {
131 if (!device->Ok()) {
132 all_device_ok_ = false;
133 for (const auto &callback : fault_offline_callbacks_) {
134 callback(device);
135 }
136 }
137 }
138 return all_device_ok_;
139 }
140
145 void OnDeviceFaultOrOffline(CallbackType callback) {
146 if (fault_offline_callbacks_.size() < 10) {
147 fault_offline_callbacks_.emplace_back(callback);
148 } else {
149 Throw(std::runtime_error("DeviceManager: Too many fault callbacks!"));
150 }
151 }
152
156 bool all_device_ok() const { return all_device_ok_; }
157
158 private:
159 bool all_device_ok_{false};
160 etl::vector<CallbackType, 10> fault_offline_callbacks_;
161 etl::vector<Device *, kMaxDevices> devices_;
162};
