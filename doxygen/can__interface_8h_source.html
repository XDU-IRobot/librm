110 void RegisterDevice(device::CanDevice &device, u16 rx_stdid) {
111 // 不允许同一个设备在同一个ID下注册多次
112 const auto target_device_array = rx_id_to_device_list_map_.find(rx_stdid);
113 if (target_device_array != rx_id_to_device_list_map_.end()) {
114 for (const auto &registered_device : rx_id_to_device_list_map_[rx_stdid]) {
115 if (&device == registered_device) {
116 return;
117 }
118 }
119 }
120
121 if (target_device_array == rx_id_to_device_list_map_.end()) {
122 rx_id_to_device_list_map_[rx_stdid] = std::vector<device::CanDevice *>{};
123 }
124 rx_id_to_device_list_map_[rx_stdid].push_back(&device);
125 }
126
130 void UnregisterDevice(device::CanDevice &device) {
131 for (auto &[rx_stdid, device_array] : rx_id_to_device_list_map_) {
132 for (auto it = device_array.begin(); it != device_array.end(); ++it) {
133 if (*it == &device) {
134 device_array.erase(it);
135 break; // 一个设备只能在一个ID下注册一次，所以找到就可以退出这一层循环了
136 }
137 }
138 }
139 }

141 const std::vector<device::CanDevice *> &GetDeviceListByRxStdid(u16 rx_stdid) const {
142 const auto target_device_array = rx_id_to_device_list_map_.find(rx_stdid);
143 if (target_device_array != rx_id_to_device_list_map_.end()) {
144 return target_device_array->second;
145 }
146 return empty_device_list_;
147 }
