55constexpr u16 CalculateNoteFreq(u16 a4_freq, i16 semitones) {
56 // 使用近似计算：2^(n/12) ≈ 使用查找表或者分段计算
57 // 为了编译期计算，使用整数运算
58 f32 ratio = 1.0f;
59 f32 semitone_ratio = 1.059463094359f; // 2^(1/12)
60
61 if (semitones > 0) {
62 for (i16 i = 0; i < semitones; ++i) {
63 ratio *= semitone_ratio;
64 }
65 } else if (semitones < 0) {
66 for (i16 i = 0; i > semitones; --i) {
67 ratio /= semitone_ratio;
68 }
69 }
70
71 return static_cast<u16>(a4_freq * ratio + 0.5f);
72}

191template <u16 Bpm>
192struct NoteDuration {
193 // 四分音符时值 = 60000ms / BPM
194 static constexpr u16 kQuarter = 60000 / Bpm;
195
196 // 基本音符时值
197 static constexpr u16 kWhole = kQuarter * 4;
198 static constexpr u16 kHalf = kQuarter * 2;
199 static constexpr u16 kEighth = kQuarter / 2;
200 static constexpr u16 kSixteenth = kQuarter / 4;
201 static constexpr u16 kThirtySecond = kQuarter / 8;
202
203 // 附点音符时值（原时值的1.5倍）
204 static constexpr u16 kDottedWhole = kWhole + kHalf;
205 static constexpr u16 kDottedHalf = kHalf + kQuarter;
206 static constexpr u16 kDottedQuarter = kQuarter + kEighth;
207 static constexpr u16 kDottedEighth = kEighth + kSixteenth;
208 static constexpr u16 kDottedSixteenth = kSixteenth + kThirtySecond;
209
210 // 三连音时值（原时值的2/3）
211 static constexpr u16 kTripletHalf = (kHalf * 2) / 3;
212 static constexpr u16 kTripletQuarter = (kQuarter * 2) / 3;
213 static constexpr u16 kTripletEighth = (kEighth * 2) / 3;
214};

252template <usize BeepsCount>
253class Beeps : public BuzzerMelody {
254 public:
255 Beeps() = default;
256
257 BuzzerNote Update(TimePoint now) override {
258 using Freq = NoteFreqStandard;
259 using Duration = NoteDuration160;
260
261 if (note_index_ >= BeepsCount * 2) {
262 return BuzzerNote(0, 0); // 播放完毕
263 }
264
265 BuzzerNote note;
266 if (note_index_ % 2 == 0) {
267 note = BuzzerNote(Freq::kE6, Duration::kThirtySecond); // 响声
268 } else {
269 note = BuzzerNote(Freq::kRest, Duration::kThirtySecond); // 静音
270 }
271
272 auto elapsed = ElapsedMs(note_start_time_, now);
273 if (elapsed >= note.duration) {
274 note_index_++;
275 note_start_time_ = now;
276 }
277
278 return note;
279 }
280
281 void Reset(TimePoint now) override {
282 note_index_ = 0;
283 note_start_time_ = now;
284 }
285
286 private:
287 usize note_index_{0};
288 TimePoint note_start_time_;
289};
