音符频率定义（基于十二平均律计算）

音符频率定义（基于十二平均律计算）

模板参数
A4FreqA4音符的基准频率（默认440Hz，国际标准音高）
注解
使用十二平均律计算所有音符频率 公式：f(n) = A4 * 2^((n-69)/12)，其中n为MIDI音符号 或者：f = A4 * 2^(semitones/12)，semitones为相对于A4的半音数
// 使用标准音高A4=440Hz
using Freq = NoteFreq<440>;
BuzzerNote note(Freq::kC5, 500);
// 使用巴洛克音高A4=415Hz
using BaroqueFreq = NoteFreq<415>;
BuzzerNote baroque_note(BaroqueFreq::kC5, 500);

constexpr u16 CalculateNoteFreq(u16 a4_freq, i16 semitones) {
// 使用近似计算：2^(n/12) ≈ 使用查找表或者分段计算
// 为了编译期计算，使用整数运算
f32 ratio = 1.0f;
f32 semitone_ratio = 1.059463094359f; // 2^(1/12)
if (semitones > 0) {
for (i16 i = 0; i < semitones; ++i) {
ratio *= semitone_ratio;
}
} else if (semitones < 0) {
for (i16 i = 0; i > semitones; --i) {
ratio /= semitone_ratio;
}
}
return static_cast<u16>(a4_freq * ratio + 0.5f);
}

template <u16 Bpm>
struct NoteDuration {
// 四分音符时值 = 60000ms / BPM
static constexpr u16 kQuarter = 60000 / Bpm;
// 基本音符时值
static constexpr u16 kWhole = kQuarter * 4;
static constexpr u16 kHalf = kQuarter * 2;
static constexpr u16 kEighth = kQuarter / 2;
static constexpr u16 kSixteenth = kQuarter / 4;
static constexpr u16 kThirtySecond = kQuarter / 8;
// 附点音符时值（原时值的1.5倍）
static constexpr u16 kDottedWhole = kWhole + kHalf;
static constexpr u16 kDottedHalf = kHalf + kQuarter;
static constexpr u16 kDottedQuarter = kQuarter + kEighth;
static constexpr u16 kDottedEighth = kEighth + kSixteenth;
static constexpr u16 kDottedSixteenth = kSixteenth + kThirtySecond;
// 三连音时值（原时值的2/3）
static constexpr u16 kTripletHalf = (kHalf * 2) / 3;
static constexpr u16 kTripletQuarter = (kQuarter * 2) / 3;
static constexpr u16 kTripletEighth = (kEighth * 2) / 3;
};

template <usize BeepsCount>
class Beeps : public BuzzerMelody {
public:
Beeps() = default;
BuzzerNote Update(TimePoint now) override {
using Freq = NoteFreqStandard;
using Duration = NoteDuration160;
if (note_index_ >= BeepsCount * 2) {
return BuzzerNote(0, 0); // 播放完毕
}
BuzzerNote note;
if (note_index_ % 2 == 0) {
note = BuzzerNote(Freq::kE6, Duration::kThirtySecond); // 响声
} else {
note = BuzzerNote(Freq::kRest, Duration::kThirtySecond); // 静音
}
auto elapsed = ElapsedMs(note_start_time_, now);
if (elapsed >= note.duration) {
note_index_++;
note_start_time_ = now;
}
return note;
}
void Reset(TimePoint now) override {
note_index_ = 0;
note_start_time_ = now;
}
private:
usize note_index_{0};
TimePoint note_start_time_;
};
