50template <RefereeRevision revision>
51class Referee : public Device {
52 private:
53 enum class DeserializeFsmState {
54 kSof,
55 kLenLsb,
56 kLenMsb,
57 kSeq,
58 kCrc8,
59 kCrc16,
60 } deserialize_fsm_state_{DeserializeFsmState::kSof};
61
62 public:
63 using RxCallback = std::function<void(u16,
64 u8
65 )>;
66
67 Referee() = default;
68
69 void operator<<(u8 data) {
70 switch (deserialize_fsm_state_) {
71 case DeserializeFsmState::kSof: {
72 if (data == kRefProtocolHeaderSof) {
73 deserialize_fsm_state_ = DeserializeFsmState::kLenLsb;
74 valid_data_so_far_[valid_data_so_far_idx_++] = data;
75 } else {
76 valid_data_so_far_idx_ = 0;
77 }
78 break;
79 }
80
81 case DeserializeFsmState::kLenLsb: {
82 data_len_this_time_ = data;
83 valid_data_so_far_[valid_data_so_far_idx_++] = data;
84 deserialize_fsm_state_ = DeserializeFsmState::kLenMsb;
85 break;
86 }
87
88 case DeserializeFsmState::kLenMsb: {
89 data_len_this_time_ |= (data << 8);
90 valid_data_so_far_[valid_data_so_far_idx_++] = data;
91
92 if (data_len_this_time_ < (kRefProtocolFrameMaxLen - kRefProtocolAllMetadataLen)) {
93 deserialize_fsm_state_ = DeserializeFsmState::kSeq;
94 } else {
95 deserialize_fsm_state_ = DeserializeFsmState::kSof;
96 valid_data_so_far_idx_ = 0;
97 }
98 break;
99 }
100
101 case DeserializeFsmState::kSeq: {
102 seq_this_time_ = data;
103 valid_data_so_far_[valid_data_so_far_idx_++] = data;
104 deserialize_fsm_state_ = DeserializeFsmState::kCrc8;
105 break;
106 }
107
108 case DeserializeFsmState::kCrc8: {
109 valid_data_so_far_[valid_data_so_far_idx_++] = data;
110
111 if (valid_data_so_far_idx_ == kRefProtocolHeaderLen) {
112 if (modules::Crc8(valid_data_so_far_.data(), kRefProtocolHeaderLen - 1, modules::CRC8_INIT) ==
113 valid_data_so_far_[4]) {
114 deserialize_fsm_state_ = DeserializeFsmState::kCrc16;
115 } else {
116 deserialize_fsm_state_ = DeserializeFsmState::kSof;
117 valid_data_so_far_idx_ = 0;
118 }
119 }
120 break;
121 }
122
123 case DeserializeFsmState::kCrc16: {
124 if (valid_data_so_far_idx_ < (kRefProtocolAllMetadataLen + data_len_this_time_)) {
125 valid_data_so_far_[valid_data_so_far_idx_++] = data;
126 }
127 if (valid_data_so_far_idx_ >= (kRefProtocolAllMetadataLen + data_len_this_time_)) {
128 deserialize_fsm_state_ = DeserializeFsmState::kSof;
129 valid_data_so_far_idx_ = 0;
130 crc16_this_time_ = (valid_data_so_far_[kRefProtocolAllMetadataLen + data_len_this_time_ - 1] << 8) |
131 valid_data_so_far_[kRefProtocolAllMetadataLen + data_len_this_time_ - 2];
132
133 if (modules::Crc16(valid_data_so_far_.data(), kRefProtocolAllMetadataLen + data_len_this_time_ - 2,
134 modules::CRC16_INIT) == crc16_this_time_) {
135 cmdid_this_time_ = (valid_data_so_far_[6] << 8) | valid_data_so_far_[5];
136
137 // 整包接收完+CRC校验通过
138 // 裁判系统仍然在线
139 ReportStatus(kOk);
140 // 把数据拷贝到反序列化缓冲区对应的结构体中
141 memcpy(reinterpret_cast<u8 *>(&deserialize_buffer_) +
142 referee_protocol_memory_map<revision>.at(cmdid_this_time_),
143 valid_data_so_far_.data() + kRefProtocolHeaderLen + kRefProtocolCmdIdLen, data_len_this_time_);
144 // 触发用户注册的回调函数
145 for (auto &cb : rx_callbacks_) {
146 if (cb) {
147 cb(cmdid_this_time_, seq_this_time_);
148 }
149 }
150 // 如果这一包的seq比上一包小，说明本次256包周期结束，计算丢包率
151 loss_rate_smooth_.add(static_cast<f32>(received_packets_) / 255.f * 100.f);
152 received_packets_ = 0;
153 }
154 }
155 break;
156 }
157
158 default: {
159 deserialize_fsm_state_ = DeserializeFsmState::kSof;
160 valid_data_so_far_idx_ = 0;
161 break;
162 }
163 }
164 }
165
166 void AttachCallback(const RxCallback &callback) { rx_callbacks_.push_back(callback); }
167
168 const RefereeProtocol<revision> &data() const { return deserialize_buffer_; }
169
170 f32 loss_rate() const { return loss_rate_smooth_.value(); }
171
172 private:
173 RefereeProtocol<revision> deserialize_buffer_{};
174 usize valid_data_so_far_idx_{0};
175 usize data_len_this_time_{0};
176 usize cmdid_this_time_{0};
177 u8 seq_this_time_{0};
178 u8 received_packets_{0};
179 u16 crc16_this_time_{0};
180 std::array<u8, kRefProtocolFrameMaxLen> valid_data_so_far_{};
181
182 std::vector<RxCallback> rx_callbacks_;
183 etl::pseudo_moving_average<f32, 10> loss_rate_smooth_{0.f};
184};
