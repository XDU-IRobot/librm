42template <typename Tp>
43static constexpr Tp max_(std::initializer_list<Tp> il) {
44 if (il.size() == 0) {
45 return 0;
46 }
47 Tp max = *il.begin();
48 for (auto &i : il) {
49 if (i > max) {
50 max = i;
51 }
52 }
53 return max;
54}

105 auto Forward(f32 vx, f32 vy, f32 w) {
106 if (vx == 0.f && vy == 0.f && w == 0.f) { // 理论上不应该直接比较浮点数，但是两个同类型浮点字面量比较应该没问题
107 forward_result_.lf_steer_position = M_PI / 4.f;
108 forward_result_.rf_steer_position = 3 * M_PI / 4.f;
109 forward_result_.lr_steer_position = 3 * M_PI / 4.f;
110 forward_result_.rr_steer_position = M_PI / 4.f;
111 } else {
112 forward_result_.lf_steer_position =
113 atan2f(vy + w * chassis_radius_ * sqrtf(2) / 2, vx + w * chassis_radius_ * sqrtf(2) / 2);
114 forward_result_.rf_steer_position =
115 atan2f(vy + w * chassis_radius_ * sqrtf(2) / 2, vx - w * chassis_radius_ * sqrtf(2) / 2);
116 forward_result_.lr_steer_position =
117 atan2f(vy - w * chassis_radius_ * sqrtf(2) / 2, vx + w * chassis_radius_ * sqrtf(2) / 2);
118 forward_result_.rr_steer_position =
119 atan2f(vy - w * chassis_radius_ * sqrtf(2) / 2, vx - w * chassis_radius_ * sqrtf(2) / 2);
120 }
121
122 forward_result_.lf_wheel_speed =
123 sqrtf(pow(vy - w * chassis_radius_ * sqrtf(2) / 2, 2) + pow(vx - w * chassis_radius_ * sqrtf(2) / 2, 2));
124 forward_result_.rf_wheel_speed =
125 sqrtf(pow(vy + w * chassis_radius_ * sqrtf(2) / 2, 2) + pow(vx - w * chassis_radius_ * sqrtf(2) / 2, 2));
126 forward_result_.lr_wheel_speed =
127 sqrtf(pow(vy + w * chassis_radius_ * sqrtf(2) / 2, 2) + pow(vx + w * chassis_radius_ * sqrtf(2) / 2, 2));
128 forward_result_.rr_wheel_speed =
129 sqrtf(pow(vy - w * chassis_radius_ * sqrtf(2) / 2, 2) + pow(vx + w * chassis_radius_ * sqrtf(2) / 2, 2));
130
131 return forward_result_;
132 }

144 auto Forward(f32 vx, f32 vy, f32 w, f32 current_lf_angle, f32 current_rf_angle, f32 current_lr_angle,
145 f32 current_rr_angle) {
146 Forward(vx, vy, w);
147
148 // 依次计算每个舵的目标角度和目前角度的差值，如果差值大于90度，就把目标舵角加180度，轮速取反。
149 // 这样可以保证舵角变化量始终小于90度，加快舵的响应速度
150 if (std::abs(Wrap(current_lf_angle - forward_result_.lf_steer_position, -M_PI, M_PI)) > M_PI / 2) {
151 forward_result_.lf_steer_position = Wrap(forward_result_.lf_steer_position + M_PI, 0, 2 * M_PI);
152 forward_result_.lf_wheel_speed = -forward_result_.lf_wheel_speed;
153 }
154 if (std::abs(Wrap(current_rf_angle - forward_result_.rf_steer_position, -M_PI, M_PI)) > M_PI / 2) {
155 forward_result_.rf_steer_position = Wrap(forward_result_.rf_steer_position + M_PI, 0, 2 * M_PI);
156 forward_result_.rf_wheel_speed = -forward_result_.rf_wheel_speed;
157 }
158 if (std::abs(Wrap(current_lr_angle - forward_result_.lr_steer_position, -M_PI, M_PI)) > M_PI / 2) {
159 forward_result_.lr_steer_position = Wrap(forward_result_.lr_steer_position + M_PI, 0, 2 * M_PI);
160 forward_result_.lr_wheel_speed = -forward_result_.lr_wheel_speed;
161 }
162 if (std::abs(Wrap(current_rr_angle - forward_result_.rr_steer_position, -M_PI, M_PI)) > M_PI / 2) {
163 forward_result_.rr_steer_position = Wrap(forward_result_.rr_steer_position + M_PI, 0, 2 * M_PI);
164 forward_result_.rr_wheel_speed = -forward_result_.rr_wheel_speed;
165 }
166
167 return forward_result_;
168 }

173 double lf_wheel_speed, rf_wheel_speed, lr_wheel_speed, rr_wheel_speed;
174 } forward_result_{};
175 double chassis_radius_;

191 auto Forward(f32 vx, f32 vy, f32 wz, bool normalize = false) {
192 forward_result_.lf_speed = vx + vy + wz;
193 forward_result_.rf_speed = vx - vy + wz;
194 forward_result_.lr_speed = -vx + vy + wz;
195 forward_result_.rr_speed = -vx - vy + wz;
196 if (normalize) {
197 const f32 max_speed = max_({std::abs(forward_result_.lf_speed), std::abs(forward_result_.rf_speed),
198 std::abs(forward_result_.lr_speed), std::abs(forward_result_.rr_speed)});
199 if (max_speed > 1) {
200 forward_result_.lf_speed /= max_speed;
201 forward_result_.rf_speed /= max_speed;
202 forward_result_.lr_speed /= max_speed;
203 forward_result_.rr_speed /= max_speed;
204 }
205 }
206
207 return forward_result_;
208 }

216 auto Inverse(f32 lf_speed, f32 rf_speed, f32 lr_speed, f32 rr_speed) {
217 inverse_result_.vx = (lf_speed + rf_speed - lr_speed - rr_speed) / 4;
218 inverse_result_.vy = (-lf_speed + rf_speed - lr_speed + rr_speed) / 4;
219 inverse_result_.wz = (lf_speed + rf_speed + lr_speed + rr_speed) / 4;
220 // normalize
221 f32 max_speed = max_({std::abs(inverse_result_.vx), std::abs(inverse_result_.vy), std::abs(inverse_result_.wz)});
222 if (max_speed > 1) {
223 inverse_result_.vx /= max_speed;
224 inverse_result_.vy /= max_speed;
225 inverse_result_.wz /= max_speed;
226 }
227
228 return inverse_result_;
229 }

231 auto inverse_result() const { return inverse_result_; }
