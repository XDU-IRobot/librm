77template <template <class> class CovarianceBase = Kalman::StandardBase>
78class SystemModel : public Kalman::LinearizedSystemModel<State, Control, CovarianceBase> {
79 public:
81 using S = rm::modules::State;
83 using C = rm::modules::Control;
84
93 explicit SystemModel(const f32 sample_freq = 1000.0f) : sample_freq_(sample_freq) {}
94
104 void GetGyro(f32 gx, f32 gy, f32 gz) {
105 gx_ = gx;
106 gy_ = gy;
107 gz_ = gz;
108 }
109
119 S f(const S &x, const C &u) const {
121 S x_;
122
123 // 中间量
124 f32 dt = 1.0f / sample_freq_;
125 f32 gx_real = gx_ - x.wx_bias();
126 f32 gy_real = gy_ - x.wy_bias();
127 f32 gz_real = gz_; // 大部分时间机体z轴通天，无法用加速度计校准
128
129 f32 q0_last = x.q0();
130 f32 q1_last = x.q1();
131 f32 q2_last = x.q2();
132 f32 q3_last = x.q3();
133
134 // 四元数预测更新
135 // - 0 -gx -gy -gz - - q0 -
136 // | gx 0 gz -gy | * | q1 | * 0.5 * dt
137 // | gy -gz 0 gx | | q2 |
138 // - gz gy -gx 0 - - q3 -
139 x_.q0() = q0_last + (-q1_last * gx_real - q2_last * gy_real - q3_last * gz_real) * dt * 0.5f;
140 x_.q1() = q1_last + (q0_last * gx_real + q2_last * gz_real - q3_last * gy_real) * dt * 0.5f;
141 x_.q2() = q2_last + (q0_last * gy_real - q1_last * gz_real + q3_last * gx_real) * dt * 0.5f;
142 x_.q3() = q3_last + (q0_last * gz_real + q1_last * gy_real - q2_last * gx_real) * dt * 0.5f;
143
144 // 陀螺仪x，y轴误差预测更新
145 x_.wx_bias() = x.wx_bias();
146 x_.wy_bias() = x.wy_bias();
147
148 return x_;
149 }
150
151 protected:
160 void updateJacobians(const S &x, const C &u) {
161 // 中间量
162 f32 dt = 1.0f / sample_freq_;
163 f32 gx_real = gx_ - x.wx_bias();
164 f32 gy_real = gy_ - x.wy_bias();
165 f32 gz_real = gz_; // 大部分时间机体z轴通天，无法用加速度计校准
166
167 f32 q0_last = x.q0();
168 f32 q1_last = x.q1();
169 f32 q2_last = x.q2();
170 f32 q3_last = x.q3();
171
172 // F = df/dx
173 this->F.setIdentity(); // 现在 F = I
174
175 Kalman::Matrix<f32, 4, 4> Fq;
176 Fq << 0, -gx_real, -gy_real, -gz_real, //
177 gx_real, 0, gz_real, -gy_real, //
178 gy_real, -gz_real, 0, gx_real, //
179 gz_real, gy_real, -gx_real, 0; //
180 this->F.template block<4, 4>(S::Q0, S::Q0) += Fq * dt * 0.5f;
181
182 Kalman::Matrix<f32, 4, 2> Fw;
183 Fw << q1_last, q2_last, //
184 -q0_last, q3_last, //
185 -q3_last, -q0_last, //
186 q2_last, -q1_last; //
187 this->F.template block<4, 2>(S::Q0, S::WX_BIAS) += Fw * dt * 0.5f;
188
189 // W = df/dw
190 this->W.setIdentity(); // TODO: 过程模型噪声协方差矩阵
191 }
192
193 private:
194 f32 sample_freq_;
195 f32 gx_, gy_, gz_;
196};

218template <template <class> class CovarianceBase = Kalman::StandardBase>
219class accelMeasurementModel : public Kalman::LinearizedMeasurementModel<State, accelMeasurement, CovarianceBase> {
220 public:
222 using S = rm::modules::State;
224 using M = rm::modules::accelMeasurement;
225
226 // 构造函数
227 accelMeasurementModel() {
228 this->H.setIdentity();
229 this->V.setIdentity();
230 }
231
232 M h(const S &x) const {
233 M measurement;
234
235 measurement << 2 * (x.q1() * x.q3() - x.q0() * x.q2()), //
236 2 * (x.q2() * x.q3() + x.q0() * x.q1()), //
237 1 - 2 * (x.q1() * x.q1() + x.q2() * x.q2()); //
238
239 return measurement;
240 }
241
242 protected:
243 void updateJacobians(const S &x) {
244 this->H.setZero();
245
246 this->H.template block<3, 4>(M::AX, S::Q0) << -2 * x.q2(), 2 * x.q3(), -2 * x.q0(), 2 * x.q1(), //
247 2 * x.q1(), 2 * x.q0(), 2 * x.q3(), 2 * x.q2(), //
248 2 * x.q0(), -2 * x.q1(), -2 * x.q2(), 2 * x.q3(); //
249
250 this->V.setIdentity();
251 }
252};
