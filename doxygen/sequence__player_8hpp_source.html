130template <typename OutputType, typename... SequenceTypes>
131class SequencePlayer {
132 static_assert(sizeof...(SequenceTypes) > 0, "At least one sequence type must be provided");
133 static_assert((std::is_base_of<SequenceGenerator<OutputType>, SequenceTypes>::value && ...),
134 "All SequenceTypes must derive from SequenceGenerator<OutputType>");
135
136 public:
137 using Output = OutputType;
138
139 SequencePlayer() : sequences_{SequenceTypes()...} {
140 // 初始化时设置当前更新函数指向第一个序列的Update方法
141 SetSequenceByIndex<0>();
142 }
143
148 template <typename Sequence>
149 void SetSequence() {
150 constexpr usize sequence_id = detail::tuple_element_index<Sequence, std::tuple<SequenceTypes...>>::value;
151 SetSequenceByIndex<sequence_id>();
152 }
153
158 Output Update() {
159 const auto now = std::chrono::steady_clock::now();
160 // 直接调用当前序列的Update方法，传入当前时间
161 current_output_ = current_update_fn_(now);
162 return current_output_;
163 }
164
169 const Output &current_output() const { return current_output_; }
170
174 void Reset() {
175 auto now = std::chrono::steady_clock::now();
176 if (current_reset_fn_) {
177 current_reset_fn_(now);
178 }
179 }
180
181 private:
186 template <usize sequence_id>
187 void SetSequenceByIndex() {
188 auto now = std::chrono::steady_clock::now();
189 // 重置序列状态
190 std::get<sequence_id>(sequences_).Reset(now);
191 // 使用etl::delegate绑定对应序列的Update方法
192 current_update_fn_ = etl::delegate<Output(std::chrono::steady_clock::time_point)>::template create<
193 SequencePlayer, &SequencePlayer::UpdateSequence<sequence_id>>(*this);
194 // 绑定Reset方法
195 current_reset_fn_ = etl::delegate<void(std::chrono::steady_clock::time_point)>::create<
196 SequencePlayer, &SequencePlayer::ResetSequence<sequence_id>>(*this);
197 }
198
205 template <usize sequence_id>
206 Output UpdateSequence(std::chrono::steady_clock::time_point now) {
207 return std::get<sequence_id>(sequences_).Update(now);
208 }
209
215 template <usize sequence_id>
216 void ResetSequence(std::chrono::steady_clock::time_point now) {
217 std::get<sequence_id>(sequences_).Reset(now);
218 }
219
220 Output current_output_{};
221 std::tuple<SequenceTypes...> sequences_;
222 // 使用etl::delegate，无动态内存分配，直接指向当前序列的Update和Reset方法
223 etl::delegate<Output(std::chrono::steady_clock::time_point)> current_update_fn_;
224 etl::delegate<void(std::chrono::steady_clock::time_point)> current_reset_fn_;
225};
