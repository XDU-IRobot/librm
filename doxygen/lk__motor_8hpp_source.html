111 LkMotor(hal::CanInterface &can, usize id, bool reversed = false)
112 : CanDevice(can, 0x140 + id), id_{id}, reversed_{reversed} {}

118 template <LkMotorType t = type, std::enable_if_t<t == LkMotorType::kMS, int> = 0>
119 void SetPower(f32 power) {
120 if (reversed_) {
121 power = -power;
122 }
123 i16 power_cmd = modules::Clamp(power, -850.f, 850.f);
124 std::memset(&tx_buffer_[1], 0, sizeof(tx_buffer_) - 1);
125 tx_buffer_[0] = Instruction::kOpenLoopControl;
126 tx_buffer_[4] = *(u8 *)(&power_cmd);
127 tx_buffer_[5] = *((u8 *)(&power_cmd) + 1);
128 can_->Write(0x140 + id_, tx_buffer_, 8);
129 }
130
135 template <LkMotorType t = type, std::enable_if_t<t != LkMotorType::kMS, int> = 0>
136 void SetTorque(f32 torque_nm) {
137 if (reversed_) {
138 torque_nm = -torque_nm;
139 }
140 i16 torque_cmd;
141 if constexpr (type == LkMotorType::kMF) {
142 torque_nm = modules::Clamp(torque_nm, -16.5f, 16.5f);
143 torque_cmd = modules::Map(torque_nm, -16.5f, 16.5f, -2048, 2048);
144 } else if constexpr (type == LkMotorType::kMG) {
145 torque_nm = modules::Clamp(torque_nm, -33.f, 33.f);
146 torque_cmd = modules::Map(torque_nm, -33.f, 33.f, -2048, 2048);
147 }
148 std::memset(&tx_buffer_[1], 0, sizeof(tx_buffer_) - 1);
149 tx_buffer_[0] = Instruction::kTorqueControl;
150 tx_buffer_[4] = *(u8 *)(&torque_cmd);
151 tx_buffer_[5] = *((u8 *)(&torque_cmd) + 1);
152 can_->Write(0x140 + id_, tx_buffer_, 8);
153 }

161 void SendInstruction(Instruction instruction) {
162 switch (instruction) {
163 case kEnable: {
164 feedback_.enabled = true;
165 break;
166 }
167 case kDisable: {
168 feedback_.enabled = false;
169 break;
170 }
171 default: {
172 break;
173 }
174 }
175 std::memset(&tx_buffer_[1], 0, sizeof(tx_buffer_) - 1);
176 tx_buffer_[0] = instruction;
177 can_->Write(0x140 + id_, tx_buffer_, 8);
178 }

184 private:
189 void RxCallback(const hal::CanFrame *msg) override {
190 ReportStatus(kOk);
191 switch (msg->data[0]) {
192 case Instruction::kOpenLoopControl: {
193 feedback_.temperature = *(i8 *)(&msg->data[1]);
194 feedback_.power = (msg->data[2] | (msg->data[3] << 8));
195 feedback_.speed_rad_ps = (msg->data[4] | (msg->data[5] << 8));
196 const u16 encoder_raw = (msg->data[6] | (msg->data[7] << 8));
197 if constexpr (encoder_bits == 15) {
198 feedback_.position_rad = modules::Map(encoder_raw, 0, 32767, 0.f, 2.f * M_PI);
199 } else if constexpr (encoder_bits == 18) {
200 feedback_.position_rad = modules::Map(encoder_raw, 0, 65535, 0.f, 2.f * M_PI);
201 }
202 break;
203 }
204 case Instruction::kTorqueControl: {
205 feedback_.temperature = *(i8 *)(&msg->data[1]);
206 const i16 iq_raw = (msg->data[2] | (msg->data[3] << 8));
207 if constexpr (type == LkMotorType::kMF) {
208 feedback_.iq = modules::Map(iq_raw, -2048, 2048, -16.5f, 16.5f);
209 } else if constexpr (type == LkMotorType::kMG) {
210 feedback_.iq = modules::Map(iq_raw, -2048, 2048, -33.f, 33.f);
211 }
212 feedback_.speed_rad_ps = (msg->data[4] | (msg->data[5] << 8));
213 const u16 encoder_raw = (msg->data[6] | (msg->data[7] << 8));
214 if constexpr (encoder_bits == 15) {
215 feedback_.position_rad = modules::Map(encoder_raw, 0, 32767, 0.f, 2.f * M_PI);
216 } else if constexpr (encoder_bits == 18) {
217 feedback_.position_rad = modules::Map(encoder_raw, 0, 65535, 0.f, 2.f * M_PI);
218 }
219 break;
220 }
221 case Instruction::kReadState1: {
222 feedback_.temperature = msg->data[1];
223 const u16 voltage_raw = (msg->data[3] | (msg->data[4] << 8));
224 feedback_.voltage = voltage_raw / 10.f;
225 *(u8 *)(&feedback_.error_state) = msg->data[7];
226 break;
227 }
228 // TODO: 其他反馈报文，暂时用不到
229 default: {
230 break;
231 }
232 }
233 }
